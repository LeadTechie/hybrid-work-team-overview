**Part A: Update markerColors.ts:**

1. Remove the hardcoded maps: `TEAM_COLORS`, `DEPARTMENT_COLORS`, `OFFICE_COLORS`
2. Keep `DEFAULT_COLOR` export for fallback cases
3. Update `getColorForEmployee()` to use colorService:
   ```typescript
   import { colorService } from '../services/colorService';

   export function getColorForEmployee(employee: Employee, colorBy: ColorByOption): string {
     switch (colorBy) {
       case 'team':
         return colorService.getColor('team', employee.team);
       case 'department':
         if (!employee.department) return DEFAULT_COLOR;
         return colorService.getColor('department', employee.department);
       case 'assignedOffice':
         if (!employee.assignedOffice) return DEFAULT_COLOR;
         return colorService.getColor('assignedOffice', employee.assignedOffice);
       default:
         return DEFAULT_COLOR;
     }
   }
   ```

**Part B: Update MapLegend.tsx:**

   const employees = useEmployeeStore((s) => s.employees);

   const legendItems = useMemo(() => {
     let values: string[];
     switch (colorBy) {
       case 'team':
